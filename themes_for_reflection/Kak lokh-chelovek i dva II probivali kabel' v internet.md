def get_repository_manifest():
    allowed = ('.txt', '.md', '.json')
    files_to_read = {}
    for file in os.listdir("."):
        if file.lower().endswith(allowed) and file not in ["main.py", "requirements.txt", "amvera.yml"]:
            try:
                with open(file, "r", encoding="utf-8") as f: 
                    # Лимит 15 000 символов на файл для защиты буфера контекста
                    files_to_read[file] = f.read()[:15000]
            except: pass
    if not files_to_read: return ""
    manifest = "\n\n[ЗАГРУЖЕННЫЕ МОДУЛИ ПАМЯТИ]:\n"
    for fname, fcontent in files_to_read.items(): manifest += f"\n--- ФАЙЛ: {fname} ---\n{fcontent}\n"
    return manifest

А чтобы победить пятиминутное проклятие хелсчеков Amvera, мы внедрили в код параллельный микро-веб-сервер прямо внутри скрипта. class AmveraKeepAliveHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.send_header('Content-type', 'application/json; charset=utf-8')
        self.end_headers()
        self.wfile.write(b'{"status": "alive"}')

def start_builtin_web_server():
    server_address = ('0.0.0.0', 8080)
    httpd = HTTPServer(server_address, AmveraKeepAliveHandler)
    logger.info("Родной веб-шлюз успешно поднят на порту 8080...")
    server_thread = threading.Thread(target=httpd.serve_forever)
    server_thread.daemon = True
    server_thread.start()

async def send_split_message(update: Update, text: str):
    max_length = 4000
    chat_id = update.effective_chat.id
    if len(text) <= max_length:
        await update.get_bot().send_message(chat_id=chat_id, text=text)
        return
    while len(text) > 0:
        if len(text) <= max_length:
            await update.get_bot().send_message(chat_id=chat_id, text=text)
            break
        split_idx = text.rfind('\n', 0, max_length)
        if split_idx == -1: 
            split_idx = max_length
        await update.get_bot().send_message(chat_id=chat_id, text=text[:split_idx])
        text = text[split_idx:].lstrip()
